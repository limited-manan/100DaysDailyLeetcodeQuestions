Question No. -> 1935. Maximum Number of Words You Can Type
Type of Question -> Easy

CODE:

class Solution {
public:
    int canBeTypedWords(string text, string brokenLetters) {
        unordered_set<char>mpp (brokenLetters.begin(), brokenLetters.end());

        int cnt  = 0;
        bool broken = false;
        for(int i = 0;i<=text.size();i++){
            if(i == text.size() || text[i] == ' '){
                if(!broken) cnt++;
                broken = false;                
            } else if(mpp.count(text[i])){
                broken = true;
            }
        }
        return cnt;
    }
};
