Question No. -> 3100. Water Bottles II
Type of Question -> Medium

CODE:

class Solution {
public:
    int maxBottlesDrunk(int numBottles, int numExchange) {
        int ans = 0;

        while(numBottles >= numExchange){
            ans += numExchange;
            numBottles -= numExchange;
            numBottles++;
            numExchange++;
        }

        ans += numBottles;
        return ans;
    }
};
