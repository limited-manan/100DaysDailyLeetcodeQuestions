(1). Question No. -> 50. Pow(x,n)
Type of Question -> Medium

CODE:

class Solution {
public:
    double myPow(double x, int n) {
        long long N = n; // for handling INT_MIN
        if(N < 0){
            x = 1/x;
            N = -N;
        }

        double result = 1.0;
        while(N > 0){
            if(N % 2 == 1){
                result *= x;
            }
            x *= x;
            N /= 2;
        }
        return result;
    }
};
